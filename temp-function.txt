    // олучение общего количества объектов
    getPropertiesCount: () => {
        return new Promise((resolve, reject) => {
            const query = 'SELECT COUNT(*) as count FROM properties WHERE archive = 0';
            db.get(query, [], (err, row) => {
                if (err) {
                    reject(err);
                } else {
                    resolve(row.count);
                }
            });
        });
    }
